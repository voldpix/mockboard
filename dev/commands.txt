Mongo queries:
show all collections
> show collections

get total collections size in MB
> db.getCollectionNames().forEach(function(collName) {
    let stats = db.getCollection(collName).stats();
    print(collName + ": " + (stats.storageSize / 1024 / 1024).toFixed(2) + " MB");
});
